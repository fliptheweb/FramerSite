layerA = new Layer()

# Modify a single style property
layerA.style["outline"] = "1px solid red"

# Modify set of style properties
layerA.style = {
	"outline": "1px solid red",
	"padding": "10px"
}

# Get a specific style property
print layerA.style["outline"]
# Output: "1px solid red"
