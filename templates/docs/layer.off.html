###### layer.off(eventName string, handler function) Stop listening for an event on this layer. For a full list of events see this overview. 
layerA = new Layer()
layerA.name = "Layer A"

clickHandler = (event, layer) ->
    print "This layer was clicked", layer.name

layerA.on(Events.Click, clickHandler)
layerA.off(Events.Click, clickHandler)
