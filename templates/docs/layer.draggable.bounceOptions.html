###### layer.draggable.bounceOptions object Options for the spring animations when momentum runs beyond constraints. 
layerA = new Layer()
layerA.draggable.enabled = true
layerA.draggable.constraints = {x:0, y:0, width:200, height:200}

# Define friction, tension and tolerance of bounce
layerA.draggable.bounceOptions = {
	friction: 40,
	tension: 200,
	tolerance: 0.0001
}
