###### scroll.contentInset object Inset for the content. This will give your content extra padding between the constraints and actual content. 
scroll = new ScrollComponent({
	width: 100, height: 100
)}

layer = new Layer({
	image: "images/bg.png"
	superLayer: scroll.content
	width: 100, height: 200
})

scroll.contentInset = {top:20, right:0, bottom:20, left: 0}
