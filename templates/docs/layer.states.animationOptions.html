###### layer.states.animationOptions object Set the animation options for the state machine. By default the global animation options will be used. 
layerA = new Layer()

layerA.states.add({
	stateA: {x:500, opacity:0.5},
	stateB: {x:200, opacity:1},
})

layerA.states.animationOptions = {
	curve: "spring(100, 10, 0)"
}

layerA.states.switch("stateB")
