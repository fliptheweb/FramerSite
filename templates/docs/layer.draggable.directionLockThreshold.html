###### layer.draggable.directionLockThreshold object The thresholds for lock directions. The x and y values represent the distance you can drag in a certain direction before it starts locking. 
layerA = new Layer()
layerA.draggable.enabled = true

# Snap horizontally after dragging 50px
# Snap vertically instantly
layerA.draggable.directionLock = true
layerA.draggable.directionLockThreshold = {x:50, y:0}
